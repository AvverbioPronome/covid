c = 1
r = len(colonne)//c
assert len(colonne) % c == 0, "Make sure to even out your picture"
fig, axes = plt.subplots(r,c,sharex=True,sharey=True,figsize=(15,r*10/c))

from itertools import cycle
axes = cycle(axes.flatten())
for a in aree:
    t = filter_area(dati, a).groupby("data").sum()
    for c in colonne:
        ax = next(axes)
        
        t.plot(y=c, ax=ax, label=a['name'], title=c)
        ax.semilogy()
